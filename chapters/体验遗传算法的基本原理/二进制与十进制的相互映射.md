import numpy as np

X_BOUND = [-3, 3]
print(X_BOUND[1])
print(X_BOUND[0])

B = 2**np.arange(16)[::-1]

print(A.dot(B))

dnaSize = 16
x = X_BOUND[0] + (X_BOUND[1] - X_BOUND[0]) / float(2 ** dnaSize - 1) * A.dot(B)
