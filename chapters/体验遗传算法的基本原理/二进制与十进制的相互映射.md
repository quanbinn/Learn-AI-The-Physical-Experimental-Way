```python
import numpy as np

print(np.arange(16))
print(np.arange(16)[::-1])

B = 2**np.arange(16)[::-1]
print(B)

A = np.array( [1,0,1,1,0,0,0,1,0,1,0,1,1,0,1,1])
print(A.dot(2**np.arange(16)))
print(A.dot(B))
```

import numpy as np

X_BOUND = [-3, 3]
print(X_BOUND[1])
print(X_BOUND[0])

B = 2**np.arange(16)[::-1]
A = np.array( [1,0,1,1,0,0,0,1,0,1,0,1,1,0,1,1])
print(A.dot(B))

dnaSize = 16
x = X_BOUND[0] + (X_BOUND[1] - X_BOUND[0]) / float(2 ** dnaSize - 1) * A.dot(B)  
print(x)
