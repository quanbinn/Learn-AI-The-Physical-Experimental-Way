```python
import numpy as np

MUTATION_RATE = 0.005

child = [1,0,0,1]
dnaSize = len(child)

if np.random.rand() < MUTATION_RATE:  
    mutatePoint = np.random.randint(0, dnaSize) 
    child[mutatePoint] = child[mutatePoint] ^ 1 
    
print(child)
```
